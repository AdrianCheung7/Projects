@typecheck
def calculate_lqri(pd: PatientData) -> float:
    """
    Takes data of patients and calculates the life quality risk index (LQRI)
    defined as age + 2*(bmi) + 15 if smoker. The higher the lQRI, the higher health risk
    imposed on the patient. 
    If any of the data is missing or invalid for patient data, patient's LQRI is deemed to be 100.0 due to it being around the average
    of all patients, thus it won't affect future average LQRI calculations.
    """
    #return [] #stub
    #template based on PatientData
    if is_invalid_patient(pd):
        return 100.0
    elif is_smoker(pd):
        return ((2*pd.bmi) + pd.age + 15)
    else:
        return ((2*pd.bmi) + pd.age)

@typecheck
def average_float(lof: List[float])-> float:
    """
    Takes a list of floats and returns the average float value
    defined as the sum of floats divided by the sum of floats in the list. If length
    of list sums to 0, function returns 0.0 because dividing by 0 is undefined.
    """
    #return 100.0 #stub
    #template based on optional
    if length(lof) == 0:
        return 0.0
    else:
        return sum_float(lof)/length(lof)
